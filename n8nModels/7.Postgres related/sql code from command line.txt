# Generating database, tables and data directly from command line
sudo useradd -m kumar
sudo passwd kumar
sudo -u postgres psql -c 'drop database if exists kumar;'
sudo -u postgres psql -c 'create database kumar;'
sudo -u postgres psql -c 'drop user if exists kumar;'
sudo -u postgres psql -c 'create user kumar;'
sudo -u postgres psql -c 'grant all privileges on database kumar to kumar;' -d kumar
sudo -u postgres psql -c "alter user kumar with encrypted password 'kumar';"
sudo -u postgres psql -c " GRANT ALL ON SCHEMA public TO kumar;" -d kumar 
sudo -u postgres psql -c " CREATE EXTENSION vector;" -d kumar
sudo su kumar

	CREATE TABLE departments (
			    		department_id int PRIMARY KEY,
			    		department_name VARCHAR(100) NOT NULL
				);
	
	CREATE TABLE employees (
				    employee_id int PRIMARY KEY,
	        	            employee_name VARCHAR(100) NOT NULL,
	                	    age float,
				    departmentId INT REFERENCES departments (department_id)
			    );
